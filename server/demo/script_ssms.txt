IF DB_ID('RoseNerdsDemo') IS NULL
BEGIN
    CREATE DATABASE [RoseNerdsDemo];
END
GO

USE [RoseNerdsDemo];
GO

IF OBJECT_ID('dbo.Person', 'U') IS NULL
BEGIN
    CREATE TABLE [Person] (
        ID int IDENTITY (1, 5) PRIMARY KEY NOT NULL,
        FName varchar(50) NOT NULL,
        LName varchar(50) NOT NULL,
        Username varchar(50) NOT NULL,
        PasswordHash varchar(512) NOT NULL,
        DOB date NULL,
        [Weight] int NULL
    );
END
GO

IF OBJECT_ID('dbo.Student', 'U') IS NULL
BEGIN
    CREATE TABLE [Student] (
        ID int PRIMARY KEY REFERENCES Person(ID) NOT NULL
    );
END
GO

IF OBJECT_ID('dbo.Trainer', 'U') IS NULL
BEGIN
    CREATE TABLE [Trainer] (
        ID int PRIMARY KEY REFERENCES Person(ID) NOT NULL
    );
END
GO

IF OBJECT_ID('dbo.Class', 'U') IS NULL
BEGIN
    CREATE TABLE [Class] (
        ID int IDENTITY (1, 1) PRIMARY KEY NOT NULL,
        Name varchar(50) NOT NULL
    );
END
GO

IF OBJECT_ID('dbo.Session', 'U') IS NULL
BEGIN
    CREATE TABLE [Session] (
        ID int IDENTITY (1, 1) PRIMARY KEY NOT NULL,
        Date date NULL,
        StartTime TIME(0) NULL,
        EndTime TIME(0) NULL,
        Location varchar(50) NULL,
        Notes varchar(500) NULL,
        Visibility bit,
        StudentID int REFERENCES Student(ID) NULL,
        ClassID int REFERENCES Class(ID) NULL
    );
END
GO

IF OBJECT_ID('dbo.Teaches', 'U') IS NULL
BEGIN
    CREATE TABLE [Teaches] (
        TrainerID int REFERENCES Trainer(ID) NOT NULL,
        ClassID int REFERENCES Class(ID) NOT NULL,
        PRIMARY KEY (TrainerID, ClassID)
    );
END
GO

IF OBJECT_ID('dbo.Exercise', 'U') IS NULL
BEGIN
    CREATE TABLE [Exercise] (
        ID int IDENTITY (1, 1) PRIMARY KEY NOT NULL,
        Name varchar(50) NOT NULL,
        Category varchar(50) NOT NULL,
        Notes varchar(50),
        Duration int NULL
    );
END
GO

IF OBJECT_ID('dbo.Set', 'U') IS NULL
BEGIN
    CREATE TABLE [Set] (
        ExerciseID int REFERENCES Exercise(ID) NOT NULL,
        SessionID int REFERENCES [Session](ID) NOT NULL,
        SetNumber int NOT NULL,
        Weight decimal(5,2) NULL,
        Reps int NULL,
        PRIMARY KEY (ExerciseID, SetNumber, SessionID)
    );
END
GO

IF OBJECT_ID('dbo.Leaderboard', 'U') IS NULL
BEGIN
    CREATE TABLE [Leaderboard] (
        ID int IDENTITY PRIMARY KEY NOT NULL,
        Name varchar(50) NOT NULL
    );
END
GO

IF OBJECT_ID('dbo.On', 'U') IS NULL
BEGIN
    CREATE TABLE [On] (
        StudentID int REFERENCES Student(ID) NOT NULL,
        LeaderboardID int REFERENCES Leaderboard(ID) NOT NULL,
        ExerciseID int REFERENCES Exercise(ID) NOT NULL,
        Rank int NULL,
        PRIMARY KEY (StudentID, LeaderboardID, ExerciseID)
    );
END
GO

IF OBJECT_ID('dbo.Logs', 'U') IS NULL
BEGIN
    CREATE TABLE [Logs] (
        ExerciseID int REFERENCES Exercise(ID) NOT NULL,
        SessionID int REFERENCES Session(ID) NOT NULL,
        IsPr bit NOT NULL,
        PRIMARY KEY (ExerciseID, SessionID)
    );
END
GO

IF OBJECT_ID('dbo.HasA', 'U') IS NULL
BEGIN
    CREATE TABLE [HasA] (
        StudentID int REFERENCES Student(ID) NOT NULL,
        ClassID int REFERENCES Class(ID) NOT NULL,
        PRIMARY KEY (StudentID, ClassID)
    );
END
GO

IF OBJECT_ID('dbo.PersonalRecord', 'U') IS NULL
BEGIN
    CREATE TABLE [PersonalRecord] (
        ID int IDENTITY (1, 1) PRIMARY KEY NOT NULL,
        Weight decimal(7, 2) NOT NULL,
        Reps int NULL,
        Duration int NULL,
        Date date NOT NULL DEFAULT CAST(GETUTCDATE() AS date)
    );
END
GO

IF OBJECT_ID('dbo.Achieves', 'U') IS NULL
BEGIN
    CREATE TABLE [Achieves] (
        StudentID int REFERENCES Student(ID) NOT NULL,
        PersonalRecordID int REFERENCES PersonalRecord(ID) NOT NULL,
        PRIMARY KEY (StudentID, PersonalRecordID)
    );
END
GO

IF OBJECT_ID('dbo.Of', 'U') IS NULL
BEGIN
    CREATE TABLE [Of] (
        PersonalRecordID int REFERENCES PersonalRecord(ID) NOT NULL,
        ExerciseID int REFERENCES Exercise(ID) NOT NULL,
        PRIMARY KEY (PersonalRecordID, ExerciseID)
    );
END
GO
